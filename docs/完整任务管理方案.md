#### 1. BCE 定时轮询服务 🔴

**目标：** 每 5 分钟扫描任务，飞书提醒

**开发内容：**
```javascript
// src/scheduler/task-scheduler.js

const cron = require('node-cron');

// 定时任务配置
cron.schedule('*/5 * * * *', async () => {
  // 1. 扫描 OCC 任务 API
  const tasks = await fetchOCCTasks();
  
  // 2. 筛选未确认任务
  const unconfirmed = tasks.filter(t => 
    t.status === 'assigned' && !t.confirmedBy
  );
  
  // 3. 发送飞书提醒
  for (const task of unconfirmed) {
    await sendFeishuReminder(task);
  }
});
```

**预计工时：** 1 小时  
**负责人：** 匠心

```javascript
// src/api/feishu-webhook.js

router.post('/webhook', async (req, res) => {
  const { event } = req.body;
  const text = event.message.content.text;
  const sender = event.message.sender_id.user_id;
  
  // 解析确认回复
  if (text.includes('收到') || text.includes('确认')) {
    const taskId = extractTaskId(event.message);
    await confirmTask(taskId, sender);
  }
  
  res.json({ success: true });
});
```
