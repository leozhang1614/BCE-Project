## 🔴 P0 高优先级功能

### 1. 飞书 Webhook 优化 - 解析"收到"回复

#### 1.1 功能说明

**目标：** 用户在飞书群回复"收到"后，系统自动确认任务

**场景：**
```
BCE 系统：📋 新任务分配 @匠心
匠心：收到
BCE 系统：✅ 任务已确认（自动）
```

#### 1.2 实现原理

```
┌─────────────────────────────────────────────────────────────────┐
│              飞书 Webhook 解析流程                               │
├─────────────────────────────────────────────────────────────────┤
│                                                                 │
│  1. 用户在飞书群回复消息                                         │
│     ↓                                                           │
│  2. 飞书服务器发送 HTTP POST 到 BCE Webhook                      │
│     ↓                                                           │
│  3. BCE 接收并解析消息内容                                       │
│     ↓                                                           │
│  4. 识别"收到"、"确认"等关键词                                   │
│     ↓                                                           │
│  5. 提取任务 ID（从引用消息或上下文）                            │
│     ↓                                                           │
│  6. 调用 BCE API 确认任务                                         │
│     ↓                                                           │
│  7. 回复确认结果到飞书群                                         │
│                                                                 │
└─────────────────────────────────────────────────────────────────┘
```

    // 1. 接收飞书推送的事件

    const message = event.message;
    
    // 2. 提取消息内容
    const text = message.content.text;  // 用户回复的文本
    const senderId = message.sender_id.user_id;  // 发送者 ID
    const chatId = message.chat_id;  // 群聊 ID
    const messageId = message.message_id;  // 消息 ID

    // 3. 解析确认关键词
    const confirmKeywords = ['收到', '确认', '好的', 'ok', '好'];
    const isConfirm = confirmKeywords.some(keyword => 
      text.toLowerCase().includes(keyword.toLowerCase())
    );
    
    if (!isConfirm) {
      // 不是确认消息，忽略
      return res.json({ success: true, message: '非确认消息，忽略' });
    }
    
    // 4. 提取任务 ID（从引用消息或上下文）

    // 方式 A：从引用消息提取
    if (message.parent_id) {
      taskId = await extractTaskIdFromParent(message.parent_id);
    }
    
    // 方式 B：从消息上下文提取（正则匹配）
    if (!taskId) {
      const taskMatch = text.match(/TASK-\d+/i);
      if (taskMatch) {
        taskId = taskMatch[0];
      }
    }
    
    // 方式 C：从最近通知的任务中提取（最近 5 分钟）
    if (!taskId) {
      taskId = await getRecentTaskId(senderId, 5 * 60 * 1000);
    }
    
    if (!taskId) {
      // 无法确定任务 ID，回复提示
      await replyFeishu(chatId, messageId, '❌ 无法确定任务 ID，请回复任务编号或在任务通知下回复');
      return res.json({ success: false, error: '无法确定任务 ID' });
    }
    
    // 5. 获取发送者姓名
    const senderName = await getUserNameById(senderId);
    
    // 6. 确认任务
    const result = await TaskService.confirmTask(taskId, senderName);
    
    if (result.success) {
      // 7. 回复确认结果
      await replyFeishu(chatId, messageId, 
        `✅ 任务已确认\n任务 ID：${taskId}\n确认人：${senderName}\n时间：${new Date().toLocaleString('zh-CN')}`
      );
      
      console.log(`[飞书 Webhook] 任务确认成功：${taskId}, 确认人：${senderName}`);
    } else {
      await replyFeishu(chatId, messageId, `❌ 任务确认失败：${result.error}`);
    }
    
    res.json({ success: true, taskId, senderName });
    
  } catch (error) {
    console.error('[飞书 Webhook] 处理失败:', error);

  }
});

### 2. 定时轮询服务

#### 2.1 功能说明

**目标：** 每 5 分钟自动扫描 OCC 任务，发现新任务发送飞书提醒

#### 2.2 实现原理

```
┌─────────────────────────────────────────────────────────────────┐

├─────────────────────────────────────────────────────────────────┤

│  │  */5 * * * *    │                                           │
│  └────────┬────────┘                                           │
│           │                                                     │

│  ┌─────────────────┐                                           │

│  ┌─────────────────┐                                           │
│  │  TaskScanner    │ 扫描服务                                    │

│           │                                                     │

│                                                                 │

```

#### 2.3 代码实现

**文件：** `src/scheduler/task-scheduler.js`

```javascript
const cron = require('node-cron');
const TaskScanner = require('../services/task-scanner');
const NotificationService = require('../services/notification-service');

// 团队成员配置
const TEAM_MEMBERS = [
  { name: '天枢', agentId: 'tianshu', enabled: true },
  { name: '匠心', agentId: 'jiangxin', enabled: true },
  { name: '司库', agentId: 'siku', enabled: true },
  { name: '执矩', agentId: 'zhiju', enabled: true },
  { name: '磐石', agentId: 'panshi', enabled: true },
  { name: '灵犀', agentId: 'lingxi', enabled: true },
  { name: '天策', agentId: 'tiance', enabled: true }
];

class TaskScheduler {
  constructor() {
    this.jobs = new Map();
    this.scanner = new TaskScanner();
    this.notifier = new NotificationService();
  }

  /**
   * 初始化所有定时任务
   */
  async init() {
    console.log('[定时任务] 初始化调度器...');
    
    for (const member of TEAM_MEMBERS) {
      if (member.enabled) {
        await this.createJob(member);
      }
    }
    
    console.log(`[定时任务] 已创建 ${this.jobs.size} 个定时任务`);
  }

  /**
   * 为单个成员创建定时任务
   */
  async createJob(member) {
    // 每 5 分钟执行一次
    const job = cron.schedule('*/5 * * * *', async () => {
      console.log(`[定时任务] ${member.name} 的任务轮询开始`);
      await this.runScan(member);
    }, {
      timezone: 'Asia/Shanghai',
      scheduled: true
    });

    this.jobs.set(member.agentId, job);
    console.log(`[定时任务] 已为 ${member.name} 创建定时任务`);

    // 启动时立即执行一次
    setTimeout(() => this.runScan(member), 5000);
  }

  /**
   * 执行单次扫描
   */
  async runScan(member) {
    try {
      const startTime = Date.now();
      
      // 1. 扫描任务
      const scanResult = await this.scanner.scan(member.agentId);
      
      // 2. 发送通知
      if (scanResult.newTasks.length > 0) {
        console.log(`[定时任务] ${member.name} 发现 ${scanResult.newTasks.length} 个新任务`);
        await this.notifier.sendNotifications(scanResult.newTasks, member);
      }
      
      // 3. 记录日志
      const duration = Date.now() - startTime;
      console.log(`[定时任务] ${member.name} 扫描完成，耗时 ${duration}ms`);
      
    } catch (error) {
      console.error(`[定时任务] ${member.name} 扫描失败:`, error);
    }
  }

  /**
   * 停止所有定时任务
   */
  async stop() {
    for (const [agentId, job] of this.jobs.entries()) {
      job.stop();
      console.log(`[定时任务] 已停止 ${agentId} 的定时任务`);
    }
    this.jobs.clear();
  }
}

// 导出单例
module.exports = new TaskScheduler();
```

**文件：** `src/services/task-scanner.js`

```javascript
const axios = require('axios');

const OCC_CONFIG = {
  baseUrl: 'http://192.168.31.187:4310',
  timeout: 10000,
  retries: 3
};

class TaskScanner {
  constructor() {
    this.axios = axios.create({
      baseURL: OCC_CONFIG.baseUrl,
      timeout: OCC_CONFIG.timeout
    });
    
    // 缓存已通知的任务 ID，避免重复通知
    this.notifiedTaskIds = new Set();
  }

  /**
   * 扫描指定成员的任务
   */
  async scan(agentId) {
    const result = {
      agentId,
      scanTime: new Date().toISOString(),
      totalTasks: 0,
      newTasks: [],
      errors: []
    };

    try {
      // 1. 从 OCC API 获取任务列表
      const tasks = await this.fetchTasks(agentId);
      result.totalTasks = tasks.length;

      // 2. 筛选未确认的新任务
      const newTasks = this.filterNewTasks(tasks, agentId);
      result.newTasks = newTasks;

      // 3. 记录扫描结果
      await this.logScanResult(result);

    } catch (error) {
      console.error(`[任务扫描] ${agentId} 扫描失败:`, error.message);
      result.errors.push(error.message);
    }

    return result;
  }

  /**
   * 从 OCC API 获取任务
   */
  async fetchTasks(agentId) {
    try {
      const response = await this.axios.get('/api/tasks', {
        params: {
          owner: agentId,
          status: 'todo,in_progress,reviewing'
        }
      });
      return response.data.tasks || [];
    } catch (error) {
      // 重试逻辑
      for (let i = 0; i < OCC_CONFIG.retries; i++) {
        try {
          await this.sleep(1000 * (i + 1));
          const response = await this.axios.get('/api/tasks');
          return response.data.tasks || [];
        } catch (retryError) {
          if (i === OCC_CONFIG.retries - 1) {
            throw retryError;
          }
        }
      }
      return [];
    }
  }

  /**
   * 筛选新任务
   */
  filterNewTasks(tasks, agentId) {
    const now = Date.now();
    const newTasks = [];

    for (const task of tasks) {
      // 检查是否是未确认的任务
      if (task.status === 'todo' && !task.confirmedBy) {
        // 检查是否已通知过
        if (!this.notifiedTaskIds.has(task.taskId)) {
          // 检查是否是最近创建的任务（30 分钟内）
          const taskTime = new Date(task.createdAt || task.updatedAt).getTime();
          if (now - taskTime < 30 * 60 * 1000) {
            newTasks.push(task);
            // 标记为已通知
            this.notifiedTaskIds.add(task.taskId);
          }
        }
      }
    }

    return newTasks;
  }

  /**
   * 记录扫描日志
   */
  async logScanResult(result) {
    const fs = require('fs').promises;
    const logFile = 'logs/scan-log.jsonl';
    const logEntry = JSON.stringify(result) + '\n';
    await fs.appendFile(logFile, logEntry);
  }

  sleep(ms) {
    return new Promise(resolve => setTimeout(resolve, ms));
  }
}

module.exports = TaskScanner;
```

### 4. OCC 任务同步

#### 4.1 功能说明

**目标：** BCE 创建的任务自动同步到 OCC

#### 4.2 代码实现

**文件：** `src/integrations/occ-sync.js`

```javascript
const axios = require('axios');

const OCC_CONFIG = {
  baseUrl: 'http://192.168.31.187:4310',
  timeout: 10000,
  retries: 3,
  syncDelay: 1000  // 同步延迟（毫秒）
};

class OCCSyncService {
  constructor() {
    this.axios = axios.create({
      baseURL: OCC_CONFIG.baseUrl,
      timeout: OCC_CONFIG.timeout
    });
    
    this.syncQueue = [];
    this.isSyncing = false;
  }

  /**
   * 同步任务到 OCC
   */
  async syncTask(bceTask) {
    // 添加到同步队列
    this.syncQueue.push({
      type: 'task',
      action: 'sync',
      data: bceTask,
      timestamp: Date.now()
    });
    
    // 触发同步
    this.processQueue();
  }

  /**
   * 处理同步队列
   */
  async processQueue() {
    if (this.isSyncing || this.syncQueue.length === 0) {
      return;
    }
    
    this.isSyncing = true;
    
    while (this.syncQueue.length > 0) {
      const item = this.syncQueue.shift();
      
      try {
        await this.executeSync(item);
        await this.sleep(OCC_CONFIG.syncDelay);
      } catch (error) {
        console.error('[OCC 同步] 失败:', error.message);
        // 失败后重新加入队列（最多 3 次）
        if (item.retries < 3) {
          item.retries = (item.retries || 0) + 1;
          this.syncQueue.push(item);
        }
      }
    }
    
    this.isSyncing = false;
  }

  /**
   * 执行单次同步
   */
  async executeSync(item) {
    if (item.type === 'task' && item.action === 'sync') {
      await this.syncTaskToOCC(item.data);
    }
  }

  /**
   * 同步任务到 OCC
   */
  async syncTaskToOCC(bceTask) {
    const occTask = this.mapBceTaskToOcc(bceTask);
    
    const response = await this.axios.post('/api/tasks', occTask);
    
    console.log(`[OCC 同步] 任务已同步：${bceTask.id} → ${response.data.taskId}`);
    
    return response.data;
  }

  /**
   * BCE 任务映射到 OCC 任务
   */
  mapBceTaskToOcc(bceTask) {
    return {
      projectId: bceTask.projectId || 'bce-project',
      taskId: bceTask.id,
      title: bceTask.title,
      description: bceTask.description,
      owner: bceTask.assignee,
      creator: bceTask.creator,
      status: this.mapBceStatusToOcc(bceTask.status),
      priority: bceTask.priority,
      dueAt: bceTask.deadline,
      createdAt: bceTask.createdAt,
      updatedAt: bceTask.updatedAt,
      metadata: {
        source: 'bce',
        bceTaskId: bceTask.id,
        syncedAt: new Date().toISOString()
      }
    };
  }

  /**
   * BCE 状态映射到 OCC 状态
   */
  mapBceStatusToOcc(bceStatus) {
    const statusMap = {
      'pending': 'todo',
      'assigned': 'todo',
      'confirmed': 'todo',
      'in_progress': 'in-progress',
      'reviewing': 'review',
      'completed': 'done',
      'cancelled': 'cancelled'
    };
    return statusMap[bceStatus] || 'todo';
  }

  sleep(ms) {
    return new Promise(resolve => setTimeout(resolve, ms));
  }
}

// 导出单例
module.exports = new OCCSyncService();
```

const https = require('https');
const feishuApps = require('../config/feishu-apps');

// 缓存 token（每个应用独立）
const appTokens = new Map();

/**
 * 获取指定应用的 tenant_access_token
 */
async function getAppToken(appName) {
  const app = feishuApps.apps[appName];
  if (!app) throw new Error(`应用未配置：${appName}`);
  
  // 检查缓存
  const cached = appTokens.get(appName);
  if (cached && Date.now() < cached.expiresAt) {
    return cached.token;
  }
  
  // 获取新 token
  const token = await fetchToken(app.appId, app.appSecret);
  appTokens.set(appName, {
    token,
    expiresAt: Date.now() + 7000 * 1000 // 2 小时
  });
  
  return token;
}

/**
 * 发送@通知（使用接收者的应用）
 */
async function sendMentionNotification(receiverName, message) {
  const app = feishuApps.apps[receiverName];
  if (!app) {
    throw new Error(`接收者应用未配置：${receiverName}`);
  }
  
  const token = await getAppToken(receiverName);
  
  // 使用接收者的应用发送消息
  const content = {
    text: message,
    at: {
      user_id: [app.userId],  // 使用正确的 user_id
      all: false
    }
  };
  
  return await sendMessage(token, FEISHU_CHAT_ID, content);
}
