#### 2. 实现多应用路由

**文件：** `BCE-Project/src/api/feishu-notify-multi.js`

```javascript
const https = require('https');
const { FEISHU_APPS } = require('../config/feishu-apps');

// 缓存 token（每个应用独立）
const appTokens = new Map();

/**
 * 获取指定应用的 tenant_access_token
 */
async function getAppToken(appName) {
  const app = FEISHU_APPS[appName];
  if (!app) throw new Error(`应用未配置：${appName}`);
  
  // 检查缓存
  const cached = appTokens.get(appName);
  if (cached && Date.now() < cached.expiresAt) {
    return cached.token;
  }
  
  // 获取新 token
  const token = await fetchToken(app.appId, app.appSecret);
  appTokens.set(appName, {
    token,
    expiresAt: Date.now() + 7000 * 1000 // 2 小时 - 5 分钟
  });
  
  return token;
}

/**
 * 发送@通知（使用接收者的应用）
 */
async function sendMentionNotification(receiverName, message) {
  const app = FEISHU_APPS[receiverName];
  if (!app) {
    throw new Error(`接收者应用未配置：${receiverName}`);
  }
  
  const token = await getAppToken(receiverName);
  
  // 使用接收者的应用发送消息
  // 这样@的 user_id 是正确的！
  const content = {
    text: message,
    at: {
      user_id: [app.userId], // 使用正确的 user_id
      all: false
    }
  };
  
  return await sendMessage(token, FEISHU_CHAT_ID, content);
}
```
