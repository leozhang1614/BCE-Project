```javascript
class RetryQueue {
  constructor() {
    this.intervalId = null;
    this.start();
  }

  start() {
    if (this.intervalId) clearInterval(this.intervalId);
    this.intervalId = setInterval(() => this.process(), 30000);
  }

  stop() {
    if (this.intervalId) {
      clearInterval(this.intervalId);
      this.intervalId = null;
    }
  }

  clear() {
    this.queue = [];
    this.stop();  // v3.2 修复
  }
}
```
