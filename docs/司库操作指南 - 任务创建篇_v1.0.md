## 任务创建方式

### 方式一：API 创建（推荐）

**调用 BCE API 创建任务：**

```javascript
// 在司库机器人代码中
const task = {
  title: '北斗智投系统 - 财务知识库',
  description: '负责财务知识库搭建，包括会计科目、财务报表等',
  creator: '司库',
  assignee: '司库',  // 自己执行
  // assignee: '匠心',  // 或者分配给匠心
  priority: 'P0',
  dueDate: '2026-03-24T17:00:00Z'
};

const response = await fetch('http://localhost:3000/api/bce/tasks', {
  method: 'POST',
  headers: {'Content-Type': 'application/json'},
  body: JSON.stringify(task)
});

const result = await response.json();
console.log('任务创建成功:', result.data.id);
```

**请求参数说明：**

| 参数 | 必填 | 说明 | 示例 |
|------|------|------|------|
| title | ✅ | 任务名称 | "北斗智投系统 - 财务知识库" |
| description | ✅ | 任务描述 | "负责财务知识库搭建" |
| creator | ✅ | 创建者 | "司库" |
| assignee | ✅ | 执行者 | "匠心"/"磐石"/"司库" |
| priority | ❌ | 优先级 | "P0"/"P1"/"P2" |
| dueDate | ❌ | 截止时间 | "2026-03-24T17:00:00Z" |

---

### 方式二：批量创建任务

**一次性创建多个任务：**

```javascript
// 定义任务列表
const tasks = [
  {
    title: '北斗智投系统 - 财务知识库',
    description: '负责财务知识库搭建',
    assignee: '司库'
  },
  {
    title: '北斗智投系统 - 数据感知模块',
    description: '负责多源数据采集',
    assignee: '磐石'
  },
  {
    title: '北斗智投系统 - 智能决策模块',
    description: '负责三维分析建议',
    assignee: '匠心'
  }
];

// 批量创建
for (const task of tasks) {
  const response = await fetch('http://localhost:3000/api/bce/tasks', {
    method: 'POST',
    headers: {'Content-Type': 'application/json'},
    body: JSON.stringify({
      ...task,
      creator: '司库',
      priority: 'P0',
      dueDate: '2026-03-24T17:00:00Z'
    })
  });
  
  const result = await response.json();
  console.log(`任务"${task.title}"创建成功，ID: ${result.data.id}`);
}
```

### 步骤 2：调用 API 创建任务

**代码示例：**

```javascript
// 司库机器人代码
async function createTask(title, description, assignee) {
  const task = {
    title: title,
    description: description,
    creator: '司库',
    assignee: assignee,
    priority: 'P0',
    dueDate: new Date(Date.now() + 24 * 60 * 60 * 1000).toISOString() // 明天同一时间
  };
  
  try {
    const response = await fetch('http://localhost:3000/api/bce/tasks', {
      method: 'POST',
      headers: {'Content-Type': 'application/json'},
      body: JSON.stringify(task)
    });
    
    if (!response.ok) {
      throw new Error(`HTTP error! status: ${response.status}`);
    }
    
    const result = await response.json();
    console.log('✅ 任务创建成功!');
    console.log('任务 ID:', result.data.id);
    console.log('任务名称:', result.data.title);
    console.log('执行者:', result.data.assignee);
    console.log('截止时间:', result.data.dueDate);
    
    return result.data;
  } catch (error) {
    console.error('❌ 任务创建失败:', error.message);
    throw error;
  }
}

// 使用示例
createTask(
  '北斗智投系统 - 财务知识库',
  '负责财务知识库搭建，包括会计科目、财务报表等',
  '司库'
);
```

## 验收任务

### 验收流程

**当执行者提交验收后：**

1. 司库收到验收通知
2. 检查交付物是否符合要求
3. 验收通过 → 提交审核给执矩
4. 验收不通过 → 驳回重新开发

---

### 验收通过

```javascript
// 验收通过
const acceptTask = {
  acceptor: '司库',
  comment: '验收通过，功能完整，代码质量良好。',
  deliverables: ['财务知识库代码', '数据库设计文档', 'API 接口文档']
};

const response = await fetch('http://localhost:3000/api/bce/tasks/task-123/accept', {
  method: 'POST',
  headers: {'Content-Type': 'application/json'},
  body: JSON.stringify(acceptTask)
});

// 验收通过后，系统自动流转到"待审核"节点，通知执矩审核
```

---

### 验收不通过

```javascript
// 验收不通过，驳回重新开发
const rejectTask = {

// 验收不通过后，系统自动回退到"开发中"节点，执行者重新开发
```

## 常见问题

### Q1: 任务创建失败怎么办？

**可能原因：**
1. BCE 服务未启动
2. API 地址错误
3. 参数格式错误

**解决方案：**
```bash
# 检查 BCE 服务状态
curl http://localhost:3000/health

# 检查参数格式
console.log(JSON.stringify(task));  // 确保是有效 JSON
```

---

### Q2: 执行者说没收到任务通知？

**可能原因：**
1. 飞书通知配置错误
2. 执行者 user_id 错误
3. 通知被屏蔽

**解决方案：**
1. 检查飞书通知配置
2. 确认执行者 user_id 正确
3. 在飞书群手动@执行者

## 📝 附录

### 完整代码示例

**司库机器人完整任务创建代码：**

```javascript
// 司库机器人 - 任务创建模块
class TaskCreator {
  constructor() {
    this.apiUrl = 'http://localhost:3000/api/bce/tasks';
    this.creator = '司库';
  }
  
  /**
   * 创建单个任务
   */
  async createTask(title, description, assignee, priority = 'P0', deadlineHours = 24) {
    const task = {
      title,
      description,
      creator: this.creator,
      assignee,
      priority,
      dueDate: new Date(Date.now() + deadlineHours * 60 * 60 * 1000).toISOString()
    };
    
    try {
      const response = await fetch(this.apiUrl, {
        method: 'POST',
        headers: {'Content-Type': 'application/json'},
        body: JSON.stringify(task)
      });
      
      const result = await response.json();
      
      if (result.success) {
        console.log(`✅ 任务"${title}"创建成功，ID: ${result.data.id}`);
        return result.data;
      } else {
        console.error(`❌ 任务创建失败：${result.error}`);
        throw new Error(result.error);
      }
    } catch (error) {
      console.error(`❌ 任务创建异常：${error.message}`);
      throw error;
    }
  }
  
  /**
   * 批量创建任务
   */
  async createTasks(taskList) {
    const results = [];
    
    for (const task of taskList) {
      try {
        const result = await this.createTask(
          task.title,
          task.description,
          task.assignee,
          task.priority || 'P0',
          task.deadlineHours || 24
        );
        results.push({success: true, task: result});
      } catch (error) {
        results.push({success: false, error: error.message, task: task});
      }
    }
    
    return results;
  }
}

// 使用示例
const creator = new TaskCreator();

// 创建单个任务
creator.createTask(
  '北斗智投系统 - 财务知识库',
  '负责财务知识库搭建',
  '司库',
  'P0',
  24
);

// 批量创建任务
const tasks = [
  {title: '财务知识库', description: '财务知识库搭建', assignee: '司库'},
  {title: '数据感知模块', description: '多源数据采集', assignee: '磐石'},
  {title: '智能决策模块', description: '三维分析建议', assignee: '匠心'}
];

creator.createTasks(tasks);
```
