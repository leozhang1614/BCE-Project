### 1.2 定时任务配置

**文件：** `BCE-Project/src/scheduler/task-scheduler.js`

```javascript
/**
 * BCE 定时任务调度器
 * 为每个团队成员创建独立的定时任务
 */

const cron = require('node-cron');
const TaskScanner = require('../services/task-scanner');
const NotificationService = require('../services/notification-service');

// 团队成员配置
const TEAM_MEMBERS = [
  { name: '天枢', agentId: 'tianshu', enabled: true },
  { name: '匠心', agentId: 'jiangxin', enabled: true },
  { name: '司库', agentId: 'siku', enabled: true },
  { name: '执矩', agentId: 'zhiju', enabled: true },
  { name: '磐石', agentId: 'panshi', enabled: true },
  { name: '灵犀', agentId: 'lingxi', enabled: true },
  { name: '天策', agentId: 'tiance', enabled: true }
];

// 定时任务配置
const SCHEDULE_CONFIG = {
  // 每 5 分钟执行一次
  cronExpression: '*/5 * * * *',
  // 时区
  timezone: 'Asia/Shanghai',
  // 是否立即执行一次
  runOnInit: true
};

class TaskScheduler {
  constructor() {
    this.jobs = new Map();
    this.scanner = new TaskScanner();
    this.notifier = new NotificationService();
  }

  /**
   * 初始化所有定时任务
   */
  async init() {
    console.log('[定时任务] 初始化调度器...');
    
    for (const member of TEAM_MEMBERS) {
      if (member.enabled) {
        await this.createJob(member);
      }
    }
    
    console.log(`[定时任务] 已创建 ${this.jobs.size} 个定时任务`);
  }

  /**
   * 为单个成员创建定时任务
   */
  async createJob(member) {
    const job = cron.schedule(SCHEDULE_CONFIG.cronExpression, async () => {
      console.log(`[定时任务] ${member.name} 的任务轮询开始`);
      await this.runScan(member);
    }, {
      timezone: SCHEDULE_CONFIG.timezone,
      scheduled: true
    });

    this.jobs.set(member.agentId, job);
    console.log(`[定时任务] 已为 ${member.name} 创建定时任务 (cron: ${SCHEDULE_CONFIG.cronExpression})`);

    // 启动时立即执行一次
    if (SCHEDULE_CONFIG.runOnInit) {
      setTimeout(() => this.runScan(member), 5000);
    }
  }

  /**
   * 执行单次扫描
   */
  async runScan(member) {
    try {
      const startTime = Date.now();
      
      // 1. 扫描任务
      const scanResult = await this.scanner.scan(member.agentId);
      
      // 2. 发送通知
      if (scanResult.newTasks.length > 0) {
        await this.notifier.sendNotifications(scanResult.newTasks, member);
      }
      
      // 3. 记录日志
      const duration = Date.now() - startTime;
      console.log(`[定时任务] ${member.name} 扫描完成，发现 ${scanResult.newTasks.length} 个新任务，耗时 ${duration}ms`);
      
    } catch (error) {
      console.error(`[定时任务] ${member.name} 扫描失败:`, error);
    }
  }

  /**
   * 停止所有定时任务
   */
  async stop() {
    console.log('[定时任务] 停止所有定时任务...');
    
    for (const [agentId, job] of this.jobs.entries()) {
      job.stop();
      console.log(`[定时任务] 已停止 ${agentId} 的定时任务`);
    }
    
    this.jobs.clear();
  }

  /**
   * 获取定时任务状态
   */
  getJobStatus() {
    const status = [];
    
    for (const [agentId, job] of this.jobs.entries()) {
      const member = TEAM_MEMBERS.find(m => m.agentId === agentId);
      status.push({
        agentId,
        name: member?.name || 'Unknown',
        running: job.running,
        lastExecution: job.lastExecution
      });
    }
    
    return status;
  }
}

// 导出单例
module.exports = new TaskScheduler();
```

### 1.3 任务扫描服务

**文件：** `BCE-Project/src/services/task-scanner.js`

```javascript
/**
 * 任务扫描服务
 * 从 OCC API 获取任务并筛选
 */

const axios = require('axios');

const OCC_CONFIG = {
  baseUrl: 'http://192.168.31.187:4310',
  timeout: 10000,
  retries: 3
};

class TaskScanner {
  constructor() {
    this.axios = axios.create({
      baseURL: OCC_CONFIG.baseUrl,
      timeout: OCC_CONFIG.timeout
    });
    
    // 缓存已通知的任务 ID，避免重复通知
    this.notifiedTaskIds = new Set();
    
    // 缓存过期时间（30 分钟）
    this.cacheTTL = 30 * 60 * 1000;
  }

  /**
   * 扫描指定成员的任务
   */
  async scan(agentId) {
    const result = {
      agentId,
      scanTime: new Date().toISOString(),
      totalTasks: 0,
      newTasks: [],
      updatedTasks: [],
      errors: []
    };

    try {
      // 1. 从 OCC API 获取任务列表
      const tasks = await this.fetchTasks(agentId);
      result.totalTasks = tasks.length;

      // 2. 筛选未确认的新任务
      const newTasks = this.filterNewTasks(tasks, agentId);
      result.newTasks = newTasks;

      // 3. 筛选状态变更的任务
      const updatedTasks = await this.filterUpdatedTasks(tasks, agentId);
      result.updatedTasks = updatedTasks;

      // 4. 清理过期缓存
      this.cleanupCache();

      // 5. 记录扫描结果
      await this.logScanResult(result);

    } catch (error) {
      console.error(`[任务扫描] ${agentId} 扫描失败:`, error.message);
      result.errors.push(error.message);
    }

    return result;
  }

  /**
   * 从 OCC API 获取任务
   */
  async fetchTasks(agentId) {
    try {
      const response = await this.axios.get('/api/tasks', {
        params: {
          owner: agentId,
          status: 'todo,in_progress,reviewing'
        }
      });

      return response.data.tasks || [];
    } catch (error) {
      // 重试逻辑
      for (let i = 0; i < OCC_CONFIG.retries; i++) {
        try {
          await this.sleep(1000 * (i + 1));
          const response = await this.axios.get('/api/tasks');
          return response.data.tasks || [];
        } catch (retryError) {
          if (i === OCC_CONFIG.retries - 1) {
            throw retryError;
          }
        }
      }
      return [];
    }
  }

  /**
   * 筛选新任务
   */
  filterNewTasks(tasks, agentId) {
    const now = Date.now();
    const newTasks = [];

    for (const task of tasks) {
      // 检查是否是未确认的任务
      if (task.status === 'todo' && !task.confirmedBy) {
        // 检查是否已通知过
        if (!this.notifiedTaskIds.has(task.taskId)) {
          // 检查是否是最近创建的任务（30 分钟内）
          const taskTime = new Date(task.createdAt || task.updatedAt).getTime();
          if (now - taskTime < this.cacheTTL) {
            newTasks.push({
              ...task,
              notifyType: 'new_task'
            });
            // 标记为已通知
            this.notifiedTaskIds.add(task.taskId);
          }
        }
      }
    }

    return newTasks;
  }

  /**
   * 筛选状态变更的任务
   */
  async filterUpdatedTasks(tasks, agentId) {
    const updatedTasks = [];
    const lastStatusMap = await this.loadLastStatusMap(agentId);

    for (const task of tasks) {
      const lastStatus = lastStatusMap.get(task.taskId);
      
      // 检查状态是否变更
      if (lastStatus && lastStatus !== task.status) {
        updatedTasks.push({
          ...task,
          previousStatus: lastStatus,
          notifyType: 'status_changed'
        });
      }
      
      // 更新状态映射
      lastStatusMap.set(task.taskId, task.status);
    }

    // 保存状态映射
    await this.saveLastStatusMap(agentId, lastStatusMap);

    return updatedTasks;
  }

  /**
   * 清理过期缓存
   */
  cleanupCache() {
    // 简单实现：限制缓存大小
    if (this.notifiedTaskIds.size > 1000) {
      const ids = Array.from(this.notifiedTaskIds);
      this.notifiedTaskIds.clear();
      // 保留最近 500 个
      ids.slice(-500).forEach(id => this.notifiedTaskIds.add(id));
    }
  }

  /**
   * 记录扫描结果
   */
  async logScanResult(result) {
    // 写入日志文件
    const fs = require('fs').promises;
    const path = require('path');
    const logFile = path.join(__dirname, '../../logs/scan-log.jsonl');
    
    const logEntry = JSON.stringify({
      timestamp: result.scanTime,
      ...result
    }) + '\n';
    
    await fs.appendFile(logFile, logEntry);
  }

  sleep(ms) {
    return new Promise(resolve => setTimeout(resolve, ms));
  }

  async loadLastStatusMap(agentId) {
    // 从文件加载或使用内存缓存
    return new Map();
  }

  async saveLastStatusMap(agentId, map) {
    // 保存到文件
  }
}

module.exports = TaskScanner;
```

### 2.2 任务更新 API

**文件：** `BCE-Project/src/api/bce-tasks.js`

```javascript
/**
 * 任务更新逻辑
 * 包含状态变更、通知触发、OCC 同步
 */

const express = require('express');
const router = express.Router();
const { v4: uuidv4 } = require('uuid');
const OCCSyncService = require('../integrations/occ-sync');
const NotificationService = require('../services/notification-service');

// 内存存储
const tasks = new Map();

/**
 * 更新任务状态
 * PUT /api/bce/tasks/:id/status
 */
router.put('/tasks/:id/status', async (req, res) => {
  try {
    const { id } = req.params;
    const { status, operator, comment } = req.body;
    
    const task = tasks.get(id);
    if (!task) {
      return res.status(404).json({ error: '任务不存在' });
    }
    
    // 1. 验证状态流转
    const validTransitions = getValidTransitions(task.status);
    if (!validTransitions.includes(status)) {
      return res.status(400).json({
        error: `无效的状态流转：${task.status} → ${status}`
      });
    }
    
    // 2. 记录状态变更历史
    const oldStatus = task.status;
    task.statusHistory.push({
      from: oldStatus,
      to: status,
      operator,
      timestamp: new Date().toISOString(),
      comment
    });
    
    // 3. 更新任务状态
    task.status = status;
    task.updatedAt = new Date().toISOString();
    
    // 4. 根据状态触发不同通知
    await triggerStatusNotifications(task, oldStatus, status, operator, comment);
    
    // 5. 同步到 OCC
    await OCCSyncService.syncTask(task);
    
    // 6. 保存
    saveData();
    
    console.log(`[任务更新] ${id} 状态变更：${oldStatus} → ${status}`);
    
    res.json({
      success: true,
      task
    });
    
  } catch (error) {
    console.error('更新任务状态失败:', error);
    res.status(500).json({ error: error.message });
  }
});

/**
 * 根据状态触发通知
 */
async function triggerStatusNotifications(task, oldStatus, newStatus, operator, comment) {
  const notifications = [];
  
  switch (newStatus) {
    case 'assigned':
      // 通知负责人
      notifications.push({
        type: 'task_assigned',
        recipient: task.assignee,
        task,
        message: `📋 新任务分配：${task.title}`
      });
      break;
      
    case 'in_progress':
      // 通知创建人
      notifications.push({
        type: 'task_started',
        recipient: task.creator,
        task,
        message: `🚀 任务已开始执行：${task.title}`
      });
      break;
      
    case 'reviewing':
      // 通知验收人
      notifications.push({
        type: 'task_submitted',
        recipient: task.creator,
        task,
        message: `✅ 待验收：${task.title}`
      });
      break;
      
    case 'completed':
      // 通知负责人
      notifications.push({
        type: 'task_completed',
        recipient: task.assignee,
        task,
        message: `🎉 任务已完成：${task.title}`
      });
      break;
      
    case 'rejected':
      // 通知负责人（驳回）
      notifications.push({
        type: 'task_rejected',
        recipient: task.assignee,
        task,
        message: `❌ 任务被驳回：${task.title}`
      });
      break;
  }
  
  // 发送所有通知
  for (const notification of notifications) {
    await NotificationService.send(notification);
  }
}

/**
 * 获取有效的状态流转
 */
function getValidTransitions(fromStatus) {
  const transitions = {
    'pending': ['assigned', 'cancelled'],
    'assigned': ['confirmed', 'pending', 'cancelled'],
    'confirmed': ['in_progress', 'pending'],
    'in_progress': ['reviewing', 'blocked'],
    'reviewing': ['completed', 'rejected', 'in_progress'],
    'completed': [],
    'rejected': ['in_progress'],
    'blocked': ['in_progress', 'cancelled'],
    'cancelled': []
  };
  
  return transitions[fromStatus] || [];
}
```

### 3.2 OCC 同步服务

**文件：** `BCE-Project/src/integrations/occ-sync.js`

```javascript
/**
 * OCC 同步服务
 * 负责 BCE 与 OCC 之间的数据同步
 */

const axios = require('axios');

const OCC_CONFIG = {
  baseUrl: 'http://192.168.31.187:4310',
  timeout: 10000,
  retries: 3,
  syncDelay: 1000  // 同步延迟（毫秒）
};

class OCCSyncService {
  constructor() {
    this.axios = axios.create({
      baseURL: OCC_CONFIG.baseUrl,
      timeout: OCC_CONFIG.timeout
    });
    
    // 同步队列
    this.syncQueue = [];
    
    // 是否正在同步
    this.isSyncing = false;
  }

  /**
   * 同步任务到 OCC
   */
  async syncTask(bceTask) {
    // 添加到同步队列
    this.syncQueue.push({
      type: 'task',
      action: 'sync',
      data: bceTask,
      timestamp: Date.now()
    });
    
    // 触发同步
    this.processQueue();
  }

  /**
   * 处理同步队列
   */
  async processQueue() {
    if (this.isSyncing || this.syncQueue.length === 0) {
      return;
    }
    
    this.isSyncing = true;
    
    while (this.syncQueue.length > 0) {
      const item = this.syncQueue.shift();
      
      try {
        await this.executeSync(item);
        await this.sleep(OCC_CONFIG.syncDelay);
      } catch (error) {
        console.error('[OCC 同步] 失败:', error.message);
        // 失败后重新加入队列（最多 3 次）
        if (item.retries < 3) {
          item.retries = (item.retries || 0) + 1;
          this.syncQueue.push(item);
        }
      }
    }
    
    this.isSyncing = false;
  }

  /**
   * 执行单次同步
   */
  async executeSync(item) {
    switch (item.type) {
      case 'task':
        if (item.action === 'sync') {
          await this.syncTaskToOCC(item.data);
        } else if (item.action === 'update') {
          await this.updateTaskInOCC(item.data);
        } else if (item.action === 'delete') {
          await this.deleteTaskFromOCC(item.data);
        }
        break;
        
      case 'notification':
        await this.syncNotificationToOCC(item.data);
        break;
    }
  }

  /**
   * 同步任务到 OCC
   */
  async syncTaskToOCC(bceTask) {
    try {
      const occTask = this.mapBceTaskToOcc(bceTask);
      
      const response = await this.axios.post('/api/tasks', occTask);
      
      console.log(`[OCC 同步] 任务已同步：${bceTask.id} → ${response.data.taskId}`);
      
      return response.data;
    } catch (error) {
      console.error(`[OCC 同步] 同步任务失败：${bceTask.id}`, error.message);
      throw error;
    }
  }

  /**
   * 更新 OCC 中的任务
   */
  async updateTaskInOCC(bceTask) {
    try {
      const occTask = this.mapBceTaskToOcc(bceTask);
      
      const response = await this.axios.put(`/api/tasks/${bceTask.occTaskId}`, occTask);
      
      console.log(`[OCC 同步] 任务已更新：${bceTask.id}`);
      
      return response.data;
    } catch (error) {
      console.error(`[OCC 同步] 更新任务失败：${bceTask.id}`, error.message);
      throw error;
    }
  }

  /**
   * 从 OCC 删除任务
   */
  async deleteTaskFromOCC(bceTask) {
    try {
      await this.axios.delete(`/api/tasks/${bceTask.occTaskId}`);
      
      console.log(`[OCC 同步] 任务已删除：${bceTask.id}`);
    } catch (error) {
      console.error(`[OCC 同步] 删除任务失败：${bceTask.id}`, error.message);
      throw error;
    }
  }

  /**
   * BCE 任务映射到 OCC 任务
   */
  mapBceTaskToOcc(bceTask) {
    return {
      projectId: bceTask.projectId || 'bce-project',
      taskId: bceTask.id,
      title: bceTask.title,
      description: bceTask.description,
      owner: bceTask.assignee,
      creator: bceTask.creator,
      status: this.mapBceStatusToOcc(bceTask.status),
      priority: bceTask.priority,
      dueAt: bceTask.deadline,
      createdAt: bceTask.createdAt,
      updatedAt: bceTask.updatedAt,
      sessionKeys: bceTask.sessionKeys || [],
      metadata: {
        source: 'bce',
        bceTaskId: bceTask.id,
        syncedAt: new Date().toISOString()
      }
    };
  }

  /**
   * BCE 状态映射到 OCC 状态
   */
  mapBceStatusToOcc(bceStatus) {
    const statusMap = {
      'pending': 'todo',
      'assigned': 'todo',
      'confirmed': 'todo',
      'in_progress': 'in-progress',
      'reviewing': 'review',
      'completed': 'done',
      'rejected': 'blocked',
      'blocked': 'blocked',
      'cancelled': 'cancelled'
    };
    
    return statusMap[bceStatus] || 'todo';
  }

  /**
   * 同步通知到 OCC
   */
  async syncNotificationToOCC(notification) {
    try {
      await this.axios.post('/api/action-queue', {
        itemId: `bce:${notification.taskId}:${notification.type}`,
        taskId: notification.taskId,
        taskTitle: notification.task.title,
        type: notification.type,
        level: 'action-required',
        assignee: notification.recipient,
        createdAt: new Date().toISOString(),
        status: 'pending',
        metadata: {
          source: 'bce',
          message: notification.message
        }
      });
      
      console.log(`[OCC 同步] 通知已同步：${notification.taskId}`);
    } catch (error) {
      console.error(`[OCC 同步] 同步通知失败:`, error.message);
      throw error;
    }
  }

  sleep(ms) {
    return new Promise(resolve => setTimeout(resolve, ms));
  }
}

// 导出单例
module.exports = new OCCSyncService();
```
